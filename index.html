// AI Voice function jo score ko aawaz mein badlega function speakScore(currentScore) { // Agar browser speech support karta hai if ('speechSynthesis' in window) { // AI ko batana ke kya bolna hai let speech = new SpeechSynthesisUtterance(); speech.text = currentScore; // Jaise: "India 125 for 3" speech.lang = 'en-US'; // English commentary ke liye speech.rate = 1.0; // Bolne ki speed speech.pitch = 1.1; // Aawaz ki tone (thodi energetic) // AI bolna shuru karegi window.speechSynthesis.speak(speech); } } // Example: Jab bhi aapki score wali API update ho, aap is function ko call kar dein: // speakScore("India is 125 for 3 in 15 overs");
