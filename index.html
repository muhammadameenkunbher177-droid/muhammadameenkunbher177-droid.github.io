AI Voice Logic: Agar score badla hai, sirf tabhi AI bole if (currentScoreText !== lastSpokenScore) { speakScore(currentScoreText); lastSpokenScore = currentScoreText; // Naye score ko save kar lein } } // Bolne wala main function function speakScore(textToSpeak) { if ('speechSynthesis' in window) { // Agar pehle se AI bol rahi hai, toh use pehle rokein (taake naya score foran aaye) window.speechSynthesis.cancel(); let speech = new SpeechSynthesisUtterance(textToSpeak); speech.lang = 'en-US'; speech.rate = 1.0; speech.pitch = 1.0; window.speechSynthesis.speak(speech); } }
